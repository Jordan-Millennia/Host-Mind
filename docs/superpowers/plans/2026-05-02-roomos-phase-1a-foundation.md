# RoomOS Phase 1A — Foundation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stand up the deployed shell of RoomOS — a Next.js 14 monorepo with the full Postgres schema, Clerk-backed team auth, the brand-correct visual baseline, and an empty dashboard accessible only when signed in. Produces a Railway-deployed app that signs Jordan in, lands him on a "no data yet" home view, and has every Phase-1 table waiting for the scraper to fill them.

**Architecture:** pnpm workspace monorepo — `apps/web` (Next.js 14 App Router) + `packages/db` (Prisma schema + client). Postgres on Railway. Clerk for auth (organizations + RBAC). Tailwind v4 + shadcn/ui themed to the CoHost cream-and-gold brand. Server-side `requireRole()` middleware gates every Server Action and route handler. No scraper, no worker, no real data — those land in plans 1B–1D.

**Tech Stack:** Next.js 14, TypeScript 5, Tailwind v4, shadcn/ui, Prisma 5, PostgreSQL 16, Clerk, Vitest, Playwright (for one auth-gating e2e test), GitHub Actions, Railway.

## Source spec

`docs/superpowers/specs/2026-05-02-roomos-phase-1-design.md` — read sections 3 (architecture), 4 (data model), 7 (auth), 13 (visual direction). This plan covers the subset of section 12's "what's in Phase 1" that does NOT require the scraper or live data.

## What this plan does NOT cover (deferred to later 1B/1C/1D plans)

- The Mac Studio worker, BullMQ jobs, launchd agent, any Playwright code → **Plan 1B**
- Live dashboard with real data, KPI strip, status sections, all-rooms table, room detail → **Plan 1C**
- Bootstrap wizard (connect PadSplit, run discovery, map owners, invite team) → **Plan 1D**

## Conventions

- **Code style**: Prettier defaults + ESLint with `next/core-web-vitals` + `@typescript-eslint/strict`. No semicolons (matches existing HostMind/cohost-website code).
- **Imports**: `@roomos/db` for the database package (workspace alias). No deep imports across packages.
- **Tests**: Vitest for unit; one Playwright e2e for the auth gate. TDD order: failing test → minimal pass → refactor.
- **Commits**: small, focused, conventional-commits-ish but not strict. Each task ends with a commit.

```prisma
// RoomOS — Postgres schema (Phase 1)
// Source of truth: docs/superpowers/specs/2026-05-02-roomos-phase-1-design.md §4

generator client {
  provider        = "prisma-client-js"
  previewFeatures = []
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// =====================================================================
// Enums
// =====================================================================

enum Platform {
  PADSPLIT
  AIRBNB
  TURBOTENANT
}

enum Role {
  ADMIN
  AGENT
  OWNER  // unused in Phase 1; reserved for owner portal
}

enum OccupancyStatus {
  OCCUPIED
  MOVING_IN
  MOVING_OUT
  VACANT
  NEEDS_FLIP
  WAITING_APPROVAL
  INACTIVE
}

enum SessionStatus {
  ACTIVE
  EXPIRED
  FAILED
}

enum SyncKind {
  DISCOVERY
  OCCUPANCY
  FINANCIAL
  INTERACTIVE_LOGIN
}

enum SyncRunStatus {
  RUNNING
  SUCCESS
  PARTIAL
  FAILED
}

enum PaymentEventType {
  PAYMENT
  ADJUSTMENT
}

// =====================================================================
// Tenancy
// =====================================================================

model Org {
  id        String   @id @default(cuid())
  name      String
  createdAt DateTime @default(now()) @map("created_at")

  owners      Owner[]
  properties  Property[]
  rooms       Room[]
  listings    PlatformListing[]
  members     Member[]
  occupancies Occupancy[]
  payments    PaymentEvent[]
  teamUsers   TeamUser[]
  syncRuns    SyncRun[]
  auditLogs   AuditLog[]

  @@map("orgs")
}

model Owner {
  id              String   @id @default(cuid())
  orgId           String   @map("org_id")
  name            String
  email           String?
  phone           String?
  billingTerms    String?  @map("billing_terms")
  statementEmail  String?  @map("statement_email")
  notes           String?
  createdAt       DateTime @default(now()) @map("created_at")
  updatedAt       DateTime @updatedAt @map("updated_at")

  org        Org        @relation(fields: [orgId], references: [id], onDelete: Cascade)
  properties Property[]
  teamUsers  TeamUser[]

  @@index([orgId])
  @@map("owners")
}

// =====================================================================
// Portfolio
// =====================================================================

model Property {
  id        String   @id @default(cuid())
  orgId     String   @map("org_id")
  ownerId   String?  @map("owner_id")  // nullable until mapped during bootstrap
  name      String?
  address   String
  city      String?
  state     String?
  zip       String?
  market    String?
  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  org   Org    @relation(fields: [orgId], references: [id], onDelete: Cascade)
  owner Owner? @relation(fields: [ownerId], references: [id], onDelete: SetNull)
  rooms Room[]

  @@index([orgId])
  @@index([ownerId])
  @@map("properties")
}

model Room {
  id           String   @id @default(cuid())
  orgId        String   @map("org_id")
  propertyId   String   @map("property_id")
  name         String?
  roomNumber   String?  @map("room_number")
  maxOccupancy Int      @default(1) @map("max_occupancy")
  createdAt    DateTime @default(now()) @map("created_at")
  updatedAt    DateTime @updatedAt @map("updated_at")

  org      Org              @relation(fields: [orgId], references: [id], onDelete: Cascade)
  property Property         @relation(fields: [propertyId], references: [id], onDelete: Cascade)
  listings PlatformListing[]

  @@index([orgId])
  @@index([propertyId])
  @@map("rooms")
}

model PlatformListing {
  id                  String         @id @default(cuid())
  orgId               String         @map("org_id")
  roomId              String         @map("room_id")
  platform            Platform
  externalListingId   String?        @map("external_listing_id")
  externalPropertyId  String?        @map("external_property_id")
  isActive            Boolean        @default(true) @map("is_active")
  lastSyncedAt        DateTime?      @map("last_synced_at")
  sessionStatus       SessionStatus  @default(ACTIVE) @map("session_status")
  createdAt           DateTime       @default(now()) @map("created_at")
  updatedAt           DateTime       @updatedAt @map("updated_at")

  org         Org          @relation(fields: [orgId], references: [id], onDelete: Cascade)
  room        Room         @relation(fields: [roomId], references: [id], onDelete: Cascade)
  occupancies Occupancy[]

  @@unique([roomId, platform])
  @@index([orgId, platform])
  @@map("platform_listings")
}

// =====================================================================
// People (members) — distinct from team_users
// =====================================================================

model Member {
  id               String   @id @default(cuid())
  orgId            String   @map("org_id")
  platform         Platform
  externalMemberId String   @map("external_member_id")
  name             String
  email            String?
  phone            String?
  photoUrl         String?  @map("photo_url")
  profileUrl       String?  @map("profile_url")
  firstSeenAt      DateTime @default(now()) @map("first_seen_at")
  createdAt        DateTime @default(now()) @map("created_at")
  updatedAt        DateTime @updatedAt @map("updated_at")

  org           Org             @relation(fields: [orgId], references: [id], onDelete: Cascade)
  occupancies   Occupancy[]
  paymentEvents PaymentEvent[]

  @@unique([platform, externalMemberId])
  @@index([orgId])
  @@map("members")
}

// =====================================================================
// Occupancy & financial state
// =====================================================================

model Occupancy {
  id                   String          @id @default(cuid())
  orgId                String          @map("org_id")
  listingId            String          @map("listing_id")
  memberId             String?         @map("member_id")
  status               OccupancyStatus
  moveInDate           DateTime?       @map("move_in_date") @db.Date
  leaseEndDate         DateTime?       @map("lease_end_date") @db.Date

  // Denormalized financial state for fast dashboard reads
  currentBalance       Decimal?        @map("current_balance") @db.Decimal(10, 2)
  daysPastDue          Int?            @map("days_past_due")
  lastPaymentAt        DateTime?       @map("last_payment_at")
  lastPaymentAmount    Decimal?        @map("last_payment_amount") @db.Decimal(10, 2)
  lastFinancialSyncAt  DateTime?       @map("last_financial_sync_at")

  scrapedAt            DateTime?       @map("scraped_at")
  createdAt            DateTime        @default(now()) @map("created_at")
  updatedAt            DateTime        @updatedAt @map("updated_at")

  org           Org              @relation(fields: [orgId], references: [id], onDelete: Cascade)
  listing       PlatformListing  @relation(fields: [listingId], references: [id], onDelete: Cascade)
  member        Member?          @relation(fields: [memberId], references: [id], onDelete: SetNull)
  paymentEvents PaymentEvent[]

  @@index([orgId, status])
```

Create `apps/web/src/lib/webhook-verify.ts`:
```typescript
import { Webhook } from "svix"
import { prisma } from "@roomos/db"
import type { WebhookEvent } from "@clerk/nextjs/server"
import { env } from "./env"

/**
 * Verify a Clerk webhook signature against `CLERK_WEBHOOK_SECRET`.
 * Returns the parsed event; throws on signature failure.
 */
export function verifyClerkWebhook(headers: Headers, rawBody: string): WebhookEvent {
  const svixId = headers.get("svix-id")
  const svixTimestamp = headers.get("svix-timestamp")
  const svixSignature = headers.get("svix-signature")

  if (!svixId || !svixTimestamp || !svixSignature) {
    throw new Error("Missing svix headers")
  }

  const wh = new Webhook(env.CLERK_WEBHOOK_SECRET)
  return wh.verify(rawBody, {
    "svix-id": svixId,
    "svix-timestamp": svixTimestamp,
    "svix-signature": svixSignature,
  }) as WebhookEvent
}

/**
 * Apply a Clerk webhook event to the team_users table.
 * Pure-ish: no signature verification (caller's responsibility), no HTTP.
 */
export async function handleClerkWebhook(evt: WebhookEvent): Promise<{ ok: true }> {
  const org = await prisma.org.findFirst({ where: { name: "CoHost Management" } })
  if (!org) throw new Error("CoHost Management org not seeded — run pnpm db:seed")

  switch (evt.type) {
    case "user.created":
    case "user.updated": {
      const user = evt.data
      const email = user.email_addresses?.[0]?.email_address ?? ""
      await prisma.teamUser.upsert({
        where: { clerkUserId: user.id },
        create: {
          orgId: org.id,
          clerkUserId: user.id,
          email,
          role: "AGENT",
        },
        update: { email },
      })
      return { ok: true }
    }
    case "user.deleted": {
      const user = evt.data
      if (user.id) {
        await prisma.teamUser.deleteMany({ where: { clerkUserId: user.id } })
      }
      return { ok: true }
    }
    default:
      return { ok: true }
  }
}
```

```typescript
import { verifyClerkWebhook, handleClerkWebhook } from "@/lib/webhook-verify"

export async function POST(req: Request) {
  const rawBody = await req.text()
  let evt
  try {
    evt = verifyClerkWebhook(req.headers, rawBody)
  } catch (err) {
    console.error("clerk-webhook: signature verification failed", err)
    return NextResponse.json({ error: "invalid signature" }, { status: 400 })

  } catch (err) {
    console.error("clerk-webhook: handler failed", err)
    return NextResponse.json({ error: "handler failed" }, { status: 500 })
  }
}
```

```typescript
import { auth as clerkAuth } from "@clerk/nextjs/server"
import { prisma } from "@roomos/db"
import type { Role } from "@roomos/db"

export type Ctx = {
  userId: string         // Clerk user id
  teamUserId: string     // RoomOS team_users.id
  orgId: string
  role: Role
  ownerId: string | null
}

/** Resolve the request's auth context. Returns null when not signed in
 *  OR when the team_user record hasn't been created yet (webhook lag). */
export async function resolveContext(): Promise<Ctx | null> {
  const { userId } = await clerkAuth()
  if (!userId) return null

  const teamUser = await prisma.teamUser.findUnique({ where: { clerkUserId: userId } })
  if (!teamUser) return null

  return {
    userId,
    teamUserId: teamUser.id,
    orgId: teamUser.orgId,
    role: teamUser.role,
    ownerId: teamUser.ownerId,
  }
}

const ROLE_RANK: Record<Role, number> = {
  AGENT: 1,
  OWNER: 1,   // owner is read-only, scoped — same rank as agent for now
  ADMIN: 10,
}

/** Throw on insufficient role. ADMIN satisfies any required role. */
export async function requireRole(required: Role): Promise<Ctx> {
  const ctx = await resolveContext()
  if (!ctx) throw new Error("unauthorized: not signed in")
  if (ROLE_RANK[ctx.role] < ROLE_RANK[required]) {
    throw new Error(`forbidden: requires role ${required}, got ${ctx.role}`)
  }
  return ctx
}

/** Like requireRole but doesn't escalate; just gate signed-in. */
export async function requireSignedIn(): Promise<Ctx> {
  const ctx = await resolveContext()
  if (!ctx) throw new Error("unauthorized: not signed in")
  return ctx
}
```
